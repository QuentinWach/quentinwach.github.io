# kinetische Energien zu Bindungsenergien
def zuEV(L):
    newList = []
    for l in L:
        B = 1486.74
        Phi = 0
        newList.append( -(l - B + Phi))
    return newList

# Fläche berechnen
def areaCalc(array):
    sum_ = 0
    for a in array:
        sum_ += a * 0.1
    return sum_

# Shirley Background
def ShirleyIter(x=[0], y=[0], start=[0,0], end=[0,0], iterations=10):    
    # constants
    I_1 = start[1]
    I_2 = end[1]
    K = I_1 - I_2
    
    # init base array for background g (constant -> linear -> constant)
    g_x = x
    g_y = []
    I_1_index = 0
    I_2_index = 0
    start_i = 0; end_i = 0
    for i in range(len(g_x)):
        if g_x[i] > start[0]:
            start_i += 1
        elif g_x[i] <= start[0] and g_x[i] <= end[0]:
            end_i += 1
    for i in range(len(g_x)):
        if g_x[i] > start[0]:
            g_y.append(I_1)
        elif g_x[i] <= start[0] and g_x[i] > end[0]:
            g_y.append(I_1 - (I_1 - I_2)*(i-start_i)/(len(g_x)-start_i-end_i))
        else:
            g_y.append(I_2)
    
    # Shirley iteration over g_y array
    for a in range(iterations):
        for i in range(len(g_x)):
            # calc A_1
            A_1_data = g_y[0:i] - y[0:i]
            A_1 = areaCalc(A_1_data)
            # calc A_2
            A_2_data = g_y[i:] - y[i:]
            A_2 = areaCalc(A_2_data)
            # change y in g_y
            g_y[i] = I_2 + K * A_2 / (A_1 + A_2)
    
    # return background arrays
    return g_x, g_y

def gauß(x, omega, mu):
    return (1/(omega * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x-mu)**2)/(omega**2))

def lorentz(x, gamma, mu):
    return gamma / (np.pi * ((x-mu)**2 + gamma**2))

def voigt(x, mu, a, gamma, omega):
    return a * lorentz(x, gamma, mu) * gauß(x, omega, mu)

from scipy.optimize import curve_fit

def combined(x, 
             mu_1, a_1, gamma_1, omega_1,
             mu_2, a_2, gamma_2, omega_2):
             #mu_3, a_3, gamma_3, omega_3):
    voigt_1 = voigt(x, mu_1, a_1, gamma_1, omega_1)
    voigt_2 = voigt(x, mu_2, a_2, gamma_2, omega_2)
    #voigt_3 = voigt(x, mu_3, a_3, gamma_3, omega_3)
    return voigt_1 + voigt_2


# do the fit with some initial values
popt, pcov = curve_fit(combined, peak_1_x[38:], (peak_1_y - U)[38:], 
                       p0=(106.4, 21200, 1.38, 5,
                           110, 10000, 1.38, 6),
                       bounds=([94, 0, 0, 0, 94, 0, 0, 0], 
                               [125, np.inf, np.inf, 6, 
                                125, np.inf, np.inf, 6]))

# exakte Werte
v = popt
print("Werte: [ Verschiebung (mu), Amplitude(a), Gamma, Omega ]\n")
print(v)

# exakter Größtfehler
v_errors = np.sqrt(np.diag(pcov))
print("\nFehler:")
print(v_errors)

# totaler Fehler
rel_diff = abs(v_errors / v)
print("\nRelative Fehler:")
print(rel_diff)

if max(rel_diff) >= 1:
    print("\nKEIN ERFOLGREICHER FIT!")
    print("Es wird eine Unsicherheit eines oder mehrerer Parameter von " + str(100* max(rel_diff)) + "% erhalten!" )

# prepare some data for a plot
x = np.linspace(120, 95, 251)
peak_1_fit_y = combined(x, *popt)

# TOTAL FIT
plt.plot(x[38:], U[38:] + peak_1_fit_y[38:], "-", color="crimson", lw=3, alpha=0.8)

# INDIVIDUAL FITS
plt.plot(x[38:], U[38:] + voigt(x, popt[0], popt[1], popt[2], popt[3])[38:], "-", lw=1.5, color="#997500")
plt.plot(x[38:], U[38:] + voigt(x, popt[4], popt[5], popt[6], popt[7])[38:], "-", lw=1.5, color="#003d99")
plt.plot(x[38:], U[38:], "--", lw=1.5, color="#00991b", alpha=0.75, zorder=20)

# DATA
plt.plot(peak_1_x, peak_1_y, 'o', lw=1, ms=2, color="black")

# AREAS
a_Peak = areaCalc(voigt(x, *popt[:4]))
a_Err = areaCalc(rel_diff[:4]* a_Peak)
b_Peak = areaCalc(voigt(x, *popt[4:]))
b_Err = areaCalc(rel_diff[4:]* b_Peak)
print(str(a_Peak) + " +- " + str(a_Err))
print(str(b_Peak) + " +- " + str(b_Err))
print("Total Area: " + str(a_Peak + b_Peak) + " +- " + str(a_Err + b_Err))
